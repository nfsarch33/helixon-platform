[Unit]

[Service]
Type=simple
WorkingDirectory=%h/Code/helixon-platform
ExecStartPre=/bin/sh -c 'export OP_SERVICE_ACCOUNT_TOKEN=$(cat %h/.config/op/service-account-token) && /home/jaslian/local/bin/secrets-bootstrap --service fleet-agent --out /run/user/%U/fleet-agent-env'
